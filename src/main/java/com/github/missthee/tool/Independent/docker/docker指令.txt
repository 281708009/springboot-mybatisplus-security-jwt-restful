docker ps       查看运行中容器
docker ps -a    查看所有容器
docker image    查看所有镜像
docker run -it -p <外部端口>:<docker端口> --name <自定义容器名称> <镜像id> -d <id>  /bin/bash   运行镜像，i输入，t交互模式
docker stop     停止容器
docker rm <id>  删除容器

操作所有
docker stop $(docker ps -aq)    容器停止
docker rm $(docker ps -aq)      容器删除
docker rmi $(docker images -q)  镜像删除
